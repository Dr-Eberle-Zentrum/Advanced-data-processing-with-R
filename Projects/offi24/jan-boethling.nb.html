library(tidyverse)
library(readxl)

Data <- read_xlsx("data.xlsx")
Data %>%
  rename(cat = countriesAndTerritories,
         population = popData2020,
         ccode=countryterritoryCode)%>%
  select(cases, deaths, cat, population, ccode)%>%
  filter(!(cat == "Iceland"| cat == "Liechtenstein"| cat == "Norway"))%>%
  dplyr::group_by(cat, ccode)%>%
  summarise(
            prop_deaths=sum(deaths, na.rm= T)/population,
            prop_cases=sum(cases, na.rm= T)/population) %>%
  distinct()%>%
  arrange(desc(prop_cases))%>%
  ungroup-> mod_Data
`summarise()` has grouped output by 'cat', 'ccode'. You can override using the `.groups` argument.
mod_Data
mod_Data %>%
  mutate(country = fct_reorder(cat, prop_cases)) %>%
  ggplot( aes(x=country, y=prop_cases, fill = prop_deaths)) +
  geom_col() +
  coord_flip() +
  xlab("") +
  ylab("Cumulative cases/population")+
  scale_fill_gradient(low="white",high="darkred")+
  labs(fill="Cumulative deaths/population")+
  theme_gray()
